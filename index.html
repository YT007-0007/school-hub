FIREBASE INITIALIZATION const firebaseConfig = { apiKey: "YOUR_API_KEY", authDomain: "YOUR_PROJECT.firebaseapp.com", projectId: "YOUR_PROJECT_ID", storageBucket: "YOUR_PROJECT.appspot.com", messagingSenderId: "YOUR_ID", appId: "YOUR_APP_ID" }; firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); // 2. EMERGENCY LISTENER (Updates every tablet in real-time) db.collection("system").doc("emergency").onSnapshot((doc) => { const overlay = document.getElementById('emergency-overlay'); if (doc.exists && doc.data().active) { overlay.style.display = 'block'; } else { overlay.style.display = 'none'; } }); // 3. SCANNER ENGINE (Records attendance to the cloud) async function scanEngine(scannedData) { const [role, id] = scannedData.split(':'); if (role === 'S') { await db.collection("students").doc(id).update({ isPresent: true, lastCheckIn: new Date().toLocaleTimeString() }); console.log("Attendance recorded for: " + id); } renderView(role, id); } // 4. MESSAGING SYSTEM (For parents and teachers) async function sendMessage(studentId, teacherId, messageText) { await db.collection("chats").add({ studentId: studentId, teacherId: teacherId, text: messageText, sender: "parent", timestamp: firebase.firestore.FieldValue.serverTimestamp() }); } // 5. EMERGENCY TRIGGER (Can be called by any authorized button) async function triggerEmergency() { await db.collection("system").doc("emergency").set({ active: true, timestamp: new Date().toISOString() }); } // 6. VIEW CONTROLLER (Switches screens based on user role) function renderView(role, id) { document.querySelectorAll('.view').forEach(v => v.style.display = 'none'); if (role === 'A') document.getElementById('view-admin').style.display = 'block'; if (role === 'T') document.getElementById('view-teacher').style.display = 'block'; if (role === 'P') document.getElementById('view-principal').style.display = 'block'; if (role === 'S') document.getElementById('view-parent').style.display = 'block'; }
